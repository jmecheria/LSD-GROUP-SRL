Reguli Firestore

Copy / Paste
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAdmin() {
      return request.auth != null && request.auth.token.email == "irimia.teodor.17@gmail.com";
    }

    function isSignedIn() {
      return request.auth != null;
    }

    match /flota/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /status/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /curse/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /rezervari/{docId} {
      allow create: if isSignedIn();
      allow read: if isAdmin() || (isSignedIn() && resource.data.uid == request.auth.uid);
      allow update, delete: if isAdmin() || (isSignedIn() && resource.data.uid == request.auth.uid);
    }
  }
}
